def test_ping_aliases
  assert_respond_to(@tcp, :ping?)
  assert_respond_to(@tcp, :pingecho)
  assert_nothing_raised{ @tcp.ping? }
  assert_nothing_raised{ @tcp.ping?(@host) }

  assert_nothing_raised{ @tcp.pingecho(@host) }
end
